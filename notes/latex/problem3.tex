\begin{Exercise}[title={1d fluid flow}, label={problem:ps1q1}
%, origin={P.Paelw}
]

Assume a velocity

\begin{equation}\label{eqn:fourierSeries:10}
u = u(y)
\end{equation}

and solve NS for constant pressure gradient $G$.

e: \theExercise

a: \theAnswer

\imageFigure{../phy454/figures/continuumL4fig1}{Uniaxial stress 0}{fig:continuumL4:continuumL4fig1}{0.3}

\end{Exercise}

\begin{Answer}[ref={problem:ps1q1}]

NS for our system takes the form

\begin{equation}\label{eqn:fourierSeries:30}
\frac{d^2 u}{dy^2} = -\frac{G}{\mu}
\end{equation}

e: \theExercise

a: \theAnswer

\imageFigure{../phy454/figures/continuumL4fig1}{Uniaxial stress}{fig:continuumL4:continuumL4fig1}{0.3}

\end{Answer}
