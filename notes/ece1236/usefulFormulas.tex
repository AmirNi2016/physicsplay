\paragraph{Periodic motion}

% T c = \lambda
\begin{equation}\label{eqn:usefulFormulas:20}
v_\phi = \frac{\omega}{\beta} = \frac{\lambda}{T}
\end{equation}
% k c = \omega:
\begin{equation}\label{eqn:usefulFormulas:40}
\beta = \frac{2 \pi}{\lambda} = \frac{\omega}{v_\phi}
\end{equation}

\paragraph{Units}

\begin{equation}\label{eqn:usefulFormulas:400}
\si{V} = \si{(kg m^2)/(s^3 A)}
\end{equation}
\begin{equation}\label{eqn:usefulFormulas:420}
\si{A} = \si{ C/s}
\end{equation}
\begin{equation}\label{eqn:usefulFormulas:100}
\Omega = \si{V/A}
\end{equation}
\begin{equation}\label{eqn:usefulFormulas:440}
\si{F} = \si{ A s /V} = \si{s}/ \Omega
\end{equation}
\begin{equation}\label{eqn:usefulFormulas:60}
\si{H} = \si{V s/A} = \Omega \si{ s}
\end{equation}
\begin{equation}\label{eqn:usefulFormulas:480}
\si{W} = \si{V s}
\end{equation}
\begin{equation}\label{eqn:usefulFormulas:80}
[\epsilon] = \si{F/m}
\end{equation}
\begin{equation}\label{eqn:usefulFormulas:140}
[\BD] = \si{C/m^2}
\end{equation}
\begin{equation}\label{eqn:usefulFormulas:160}
[\BE] = \si{V/m}
\end{equation}
\begin{equation}\label{eqn:usefulFormulas:180}
[\BH] = \si{A/m}
\end{equation}
\begin{equation}\label{eqn:usefulFormulas:200}
[\BB] = \si{Wb/m}
\end{equation}
\begin{equation}\label{eqn:usefulFormulas:220}
[\BJ] = \si{A/m^2}
\end{equation}
\begin{equation}\label{eqn:usefulFormulas:240}
[\rho] = \si{C/m^3}
\end{equation}
\begin{equation}\label{eqn:usefulFormulas:460}
[\sigma] = \si{S/m} = [\omega \epsilon] = \si{F/m/s} = \si{A/V/m} = \mho/m
\end{equation}
\begin{equation}\label{eqn:usefulFormulas:120}
[\mu] = \si{H/m}
\end{equation}

\paragraph{Consistutive relations}

\begin{equation}\label{eqn:usefulFormulas:260}
\BB = \mu \BH
\end{equation}
\begin{equation}\label{eqn:usefulFormulas:280}
\BD = \epsilon \BE
\end{equation}
\begin{equation}\label{eqn:usefulFormulas:300}
\BJ = \sigma \BE
\end{equation}

\paragraph{constants}

\begin{equation}\label{eqn:usefulFormulas:320}
c = 3 \times 10^9 \,\si{m/s}
\end{equation}
\begin{equation}\label{eqn:usefulFormulas:340}
\epsilon_0 = 8.8 \times 10^{-12} \,\si{F/m}
\end{equation}
\begin{equation}\label{eqn:usefulFormulas:360}
\mu_0 = 4 \pi \times 10^{-7} \,\si{H/m}
\end{equation}
