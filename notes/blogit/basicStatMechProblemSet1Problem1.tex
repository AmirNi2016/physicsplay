\makeproblem{Limiting form of the binomial distribution}{basicStatMech:problemSet1:1}{ 
Starting from the simple case of the binomial distribution

\begin{equation}\label{eqn:basicStatMechProblemSet1Problem1:20}
P_N(X) = 2^{-N} \frac{N!}{
\left(
\frac{N + X}{2}
\right)!
\left(
\frac{N - X}{2}
\right)!
}
\end{equation}

derive the Gaussian distribution which results when $N \gg 1$ and $\Abs{X} \ll N$.
}
\makeanswer{basicStatMech:problemSet1:1}{ 

We'll work with the logarithms of $P_N(X)$.

Note that the logarithm of the Stirling approximation takes the form

\begin{dmath}\label{eqn:basicStatMechProblemSet1Problem1:40}
\ln a! 
\approx \ln \sqrt{2\pi} + \inv{2} \ln a + a \ln a - a
=
\ln \sqrt{2\pi} + \left( a + \inv{2} \right) \ln a - a
\end{dmath}

Using this we have

\begin{dmath}\label{eqn:basicStatMechProblemSet1Problem1:60}
\ln 
\left(
\frac{N + X}{2}
\right)!
=
\ln \sqrt{2 \pi}
+
\left(
\frac{N + X}{2} + \inv{2}
\right)
\left(
\ln (N + X) - \ln 2
\right)
- \frac{N + X}{2}
=
\ln \sqrt{2 \pi}
+
\left(
\frac{N + X + 1}{2} 
\right)
\left(
\ln 
\left(
1 + \frac{X}{N}
\right)
+ \ln \frac{N}{2}
\right)
- \frac{N + X}{2}
\end{dmath}

Now we can add the logs for the denominator of $\ln P_N(X)$

\begin{dmath}\label{eqn:basicStatMechProblemSet1Problem1:80}
\ln 
\left(
\frac{N + X}{2}
\right)!
+
\ln 
\left(
\frac{N - X}{2}
\right)!
=
2 \ln \sqrt{2 \pi}
-N
+
\left(
\frac{N + X + 1}{2} 
\right)
\left(
\ln 
\left(
1 + \frac{X}{N}
\right)
+ \ln \frac{N}{2}
\right)
+\left(
\frac{N - X + 1}{2} 
\right)
\left(
\ln 
\left(
1 - \frac{X}{N}
\right)
+ \ln \frac{N}{2}
\right)
\end{dmath}

Writing $\epsilon = X/N$ and using $\ln( 1 \pm \epsilon ) \approx \pm \epsilon$ for $\epsilon \ll 1$ we have

\begin{dmath}\label{eqn:basicStatMechProblemSet1Problem1:100}
\ln 
\left(
\frac{N + X}{2}
\right)!
+
\ln 
\left(
\frac{N - X}{2}
\right)!
=
2 \ln \sqrt{2 \pi}
-N
+
\left(
\frac{N + X + 1}{2} 
\right)
\left(
\epsilon
+ \ln \frac{N}{2}
\right)
+\left(
\frac{N - X + 1}{2} 
\right)
\left(
- \epsilon
+ \ln \frac{N}{2}
\right)
\end{dmath}

}
