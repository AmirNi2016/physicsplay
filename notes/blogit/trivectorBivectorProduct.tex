I can only guess that by verify you mean to expand the product in terms of it's component grades.  That expansion takes the form

\begin{dmath}\label{eqn:trivectorBivectorProduct:20}
\begin{aligned}
\lr{ a \wedge b \wedge c } \lr{ d \wedge e }
&=
\lr{ a \wedge b \wedge c } \cdot \lr{ d \wedge e } \\
&+
\gpgradethree{ \lr{ a \wedge b \wedge c } \lr{ d \wedge e } } \\
&+
a \wedge b \wedge c \wedge d \wedge e
\end{aligned}
\end{dmath}

The first term is a vector and expands as

\begin{dmath}\label{eqn:trivectorBivectorProduct:40}
\begin{aligned}
\lr{ a \wedge b \wedge c } \cdot \lr{ d \wedge e }
&=
\lr{ \lr{ a \wedge b \wedge c } \cdot d } \cdot e \\
&=
\lr{
 a \wedge b (c \cdot d)
+b \wedge c (a \cdot d)
+c \wedge a (b \cdot d)
} \cdot e \\
&=
 a (b \cdot e) (c \cdot d)
+b (c \cdot e) (a \cdot d)
+c (a \cdot e) (b \cdot d) \\
&-(a\cdot e) b (c \cdot d)
-(b\cdot e) c (a \cdot d)
-(c\cdot e) a (b \cdot d)  \\
&=
 a \lr{ (b \cdot e) (c \cdot d) -(c\cdot e) (b \cdot d) } \\
&+b \lr{ (c \cdot e) (a \cdot d) -(a\cdot e) (c \cdot d) } \\
&+c \lr{ (a \cdot e) (b \cdot d) -(b\cdot e) (a \cdot d) }.
\end{aligned}
\end{dmath}

The second term is a trivector and expands as

\begin{dmath}\label{eqn:trivectorBivectorProduct:60}
\begin{aligned}
2 \gpgradethree{ \lr{ a \wedge b \wedge c } \lr{ d \wedge e } }
&=
\gpgradethree{ \lr{ a \wedge b \wedge c } (d e - e d)} \\
&=
\gpgradethree{
\lr{ a \wedge b \wedge c } d e
-\lr{ a \wedge b \wedge c } e d
} \\
&=
\gpgradethree{
 \lr{ a \wedge b \wedge c \wedge d } e
+\lr{ \lr{ a \wedge b \wedge c } \cdot d } e
} \\
&-\gpgradethree{
 \lr{ a \wedge b \wedge c \wedge e } d
+\lr{\lr{ a \wedge b \wedge c } \cdot e } d
} \\
&=
  \lr{ a \wedge b \wedge c \wedge d } \cdot e
+ \lr{ \lr{ a \wedge b \wedge c } \cdot d } \wedge e  \\
&- \lr{ a \wedge b \wedge c \wedge e } \cdot d
- \lr{\lr{ a \wedge b \wedge c } \cdot e } \wedge d \\
&=
% a b c d
  \lr{ a \wedge b \wedge c } d \cdot e
- \lr{ a \wedge b \wedge d } c \cdot e  \\
&+ \lr{ a \wedge c \wedge d } b \cdot e
- \lr{ b \wedge c \wedge d } a \cdot e  \\
% a b c
&+ \lr{ a \wedge b \wedge e } c \cdot d
- \lr{ a \wedge c \wedge e } b \cdot d
+ \lr{ b \wedge c \wedge e } a \cdot d \\
% a b c e
&- \lr{ a \wedge b \wedge c } e \cdot d
+ \lr{ a \wedge b \wedge e } c \cdot d  \\
&- \lr{ a \wedge c \wedge e } b \cdot d
+ \lr{ b \wedge c \wedge e } a \cdot d \\
% a b c
&- \lr{ a \wedge b \wedge d } c \cdot e
+ \lr{ a \wedge c \wedge d } b \cdot e
- \lr{ b \wedge c \wedge d } a \cdot e  \\
&=
%  \lr{ a \wedge b \wedge c } (d \cdot e - e \cdot d)
 - 2 \lr{ a \wedge b \wedge d } c \cdot e
 + 2 \lr{ a \wedge c \wedge d } b \cdot e \\
&- 2 \lr{ b \wedge c \wedge d } a \cdot e
 + 2 \lr{ a \wedge b \wedge e } c \cdot d \\
&- 2 \lr{ a \wedge c \wedge e } b \cdot d
 + 2 \lr{ b \wedge c \wedge e } a \cdot d.
\end{aligned}
\end{dmath}

Putting all the pieces together, assuming no algebra errors (this is really a task better suited for a computer algebra system), the multivector product expands as

\begin{dmath}\label{eqn:trivectorBivectorProduct:80}
\begin{aligned}
\lr{ a \wedge b \wedge c } \lr{ d \wedge e }
&=
 a \lr{ (b \cdot e) (c \cdot d) -(c\cdot e) (b \cdot d) } \\
&+b \lr{ (c \cdot e) (a \cdot d) -(a\cdot e) (c \cdot d) } \\
&+c \lr{ (a \cdot e) (b \cdot d) -(b\cdot e) (a \cdot d) } \\
&- \lr{ a \wedge b \wedge d } c \cdot e
+ \lr{ a \wedge c \wedge d } b \cdot e  \\
&- \lr{ b \wedge c \wedge d } a \cdot e
+ \lr{ a \wedge b \wedge e } c \cdot d  \\
&- \lr{ a \wedge c \wedge e } b \cdot d
+ \lr{ b \wedge c \wedge e } a \cdot d \\
&+
a \wedge b \wedge c \wedge d \wedge e
\end{aligned}
\end{dmath}
