\makeproblem{Approximating directivity.}{advancedantenna:problemSet1:3}{ 
A beam antenna has half-power beamwidths of \ang{30} and 
\ang{35} in orthogonal planes
intersecting at the maximum of the mainbeam. Determine the approximate maximum
directivity.
} % makeproblem

\makeanswer{advancedantenna:problemSet1:3}{ 

\begin{dmath}\label{eqn:advancedantennaProblemSet1Problem3:10}
D_0 
\approx \frac{4 \pi}{\Theta_{1r}\Theta_{2r}}
= \frac{4 \times 180^2 }{\pi (30)( 35)}
= 39.3.
\end{dmath}
}
