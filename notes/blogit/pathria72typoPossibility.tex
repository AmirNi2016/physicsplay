\generatetitle{possible typo in 7.2.9}
%\label{chap:pathria72typoPossibility}

It's stated in the text (2001 printing of the second edition, and also online in the third edition according to google books) that

\begin{subequations}
\begin{equation}\label{eqn:pathria72typoPossibility:20}
u'(x) dx = \frac{x^3 dx}{e^x - 1}
\end{equation}
\begin{equation}\label{eqn:pathria72typoPossibility:40}
u'(x) = \frac{\pi^2 \Hbar^3 c^3}{(k T)^4} u(x)
\end{equation}
\begin{equation}\label{eqn:pathria72typoPossibility:60}
x = \frac{\Hbar \omega}{k T},
\end{equation}
\end{subequations}

is a non-dimensionalization of

\begin{equation}\label{eqn:pathria72typoPossibility:80}
u(\omega) d\omega = \frac{\Hbar}{\pi^2 c^3} \frac{\omega^3 d\omega}{ e^{\Hbar \omega/k T} - 1}.
\end{equation}

Direct substitution shows that \eqnref{eqn:pathria72typoPossibility:40} should be

\begin{equation}\label{eqn:pathria72typoPossibility:100}
u'(x) = \frac{\pi^2 \Hbar^2 c^3}{(k T)^3} u(\omega),
\end{equation}

so that we have

\begin{dmath}\label{eqn:pathria72typoPossibility:120}
u'(x) dx
= \frac{\pi^2 \Hbar^2 c^3}{(k T)^3} u(\omega) dx
= \frac{\cancel{\pi^2} \Hbar^2 \cancel{c^3}}{(k T)^3}
\frac{\Hbar}{\cancel{\pi^2} \cancel{c^3}} \frac{\omega^3 dx}{ e^{\Hbar \omega/k T} - 1}
=
\lr{\frac{\Hbar}{k T} }^3
\frac{\omega^3 dx}{ e^{\Hbar \omega/k T} - 1}
=
\frac{x^3 dx}{ e^{x} - 1 },
\end{dmath}

as desired.
