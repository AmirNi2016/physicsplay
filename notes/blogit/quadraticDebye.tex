\makeoproblem{Quadratic Deybe phonons}{pr:quadraticDebye:1}{2013 midterm} pr B2}{
Assume a quadradic dispersion relation for the longitudinal and transverse modes

\begin{equation}\label{eqn:quadraticDebye:10}
\omega = 
\left\{
\begin{array}{l}
b_{\mathrm{L}} q^2 \\
b_{\mathrm{T}} q^2
\end{array}
\right.
.
\end{equation}

\makesubproblem{}{pr:quadraticDebye:1:a}

Find the density of states.

\makesubproblem{}{pr:quadraticDebye:1:b}

Find the Deybe frequency.

\makesubproblem{}{pr:quadraticDebye:1:c}

In terms of 

\begin{equation}\label{eqn:quadraticDebye:n}
\mathcal{I} = \int_0^\infty \frac{y^{5/2} e^{y} dy}{\lr{ e^y - 1}^2 },
\end{equation}

and $\kB \Theta = \Hbar \omega_{\mathrm{D}}$ find the specific heat for $\kB T \ll \Hbar \omega_{\mathrm{D}}$

\makesubproblem{}{pr:quadraticDebye:1:d}

Find the specific heat for $\kB T \gg \Hbar \omega_{\mathrm{D}}$.
} % makeoproblem
