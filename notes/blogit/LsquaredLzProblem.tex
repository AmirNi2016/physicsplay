\paragraph{Q: \citep{sakurai2014modern} 3.17}

Given a wave function

\begin{dmath}\label{eqn:LsquaredLzProblem:20}
\psi(r,\theta, \phi) = f(r) \lr{ x + y + 3 z },
\end{dmath}

\begin{enumerate}[(a)]
\item
Determine if this wave function is an eigenfunction of \( \BL^2 \), and the value of \( l \) if it is an eigenfunction,
\item if this wave function is an eigenfunction of \( L_z \), or the probabilities of being found in any given \( \ket{l, m} \) state,
\item the expected value of \( m \) if measured.
\end{enumerate}
