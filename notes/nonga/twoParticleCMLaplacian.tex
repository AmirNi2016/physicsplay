\begin{align}\label{eqn:bohm15:2:em1}
H = 
- \frac{\hbar^2}{2 m_1} {\spacegrad_1}^2
- \frac{\hbar^2}{2 m_2} {\spacegrad_2}^2
+ V(\Br_1 -\Br_2)
\end{align}
