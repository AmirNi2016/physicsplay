\label{chap:quadraticDebye}

\makeoproblem{Quadratic Debye phonons}{pr:quadraticDebye:1}{2013 midterm pr B2}{
Assume a quadratic dispersion relation for the longitudinal and transverse modes

\begin{equation}\label{eqn:quadraticDebye:10}
\omega = 
\left\{
\begin{array}{l}
b_{\mathrm{L}} q^2 \\
b_{\mathrm{T}} q^2
\end{array}
\right.
.
\end{equation}

\makesubproblem{}{pr:quadraticDebye:1:a}

Find the density of states.

\makesubproblem{}{pr:quadraticDebye:1:b}

Find the Debye frequency.

\makesubproblem{}{pr:quadraticDebye:1:c}

In terms of $\kB \Theta = \Hbar \omega_{\mathrm{D}}$, and

\begin{equation}\label{eqn:quadraticDebye:10}
\mathcal{I} = \int_0^\infty \frac{y^{5/2} e^{y} dy}{
\lr{ e^y - 1}^2
 },
\end{equation}

find the specific heat for $\kB T \ll \Hbar \omega_{\mathrm{D}}$.

\makesubproblem{}{pr:quadraticDebye:1:d}

Find the specific heat for $\kB T \gg \Hbar \omega_{\mathrm{D}}$.
} % makeoproblem

\makeanswer{pr:quadraticDebye:1}{

\makeSubAnswer{}{pr:quadraticDebye:1:a}

Working straight from the definition

\begin{dmath}\label{eqn:quadraticDebye:30}
Z(\omega) 
= \frac{V}{(2 \pi)^3 } \sum_{L, T} \int \frac{df_\omega}{ \Abs{ \spacegrad_\Bq \omega } }
= \frac{V}{(2 \pi)^3 } 
\lr{ \evalbar{\frac{4 \pi q^2}{2 b_{\mathrm{L}} q} }{\mathrm{L}} + \evalbar{\frac{2 \times 4 \pi q^2}{2 b_{\mathrm{T}} q} }{\mathrm{T}} }
= \frac{V}{4 \pi^2 } 
\lr{ \frac{q_{\mathrm{L}}}{b_{\mathrm{L}}} + \frac{2 q_{\mathrm{T}}}{b_{\mathrm{T}}} }.
\end{dmath}

With $q_{\mathrm{L}} = \sqrt{\omega/b_{\mathrm{L}}}$ and $q_{\mathrm{T}} = \sqrt{\omega/b_{\mathrm{T}}}$, this is

\begin{dmath}\label{eqn:quadraticDebye:50}
Z(\omega) 
= \frac{V}{4 \pi^2 } 
\lr{ \frac{1}{b_{\mathrm{L}}^{3/2}} + \frac{2}{b_{\mathrm{T}}^{3/2}} }
\sqrt{\omega}
\end{dmath}

\makeSubAnswer{}{pr:quadraticDebye:1:b}

The Debye frequency was given implicitly by

\begin{dmath}\label{eqn:quadraticDebye:70}
\int_0^{\omega_{\mathrm{D}}} Z(\omega) d\omega = 3 r N,
\end{dmath}

which gives

\begin{dmath}\label{eqn:quadraticDebye:90}
3 r N
=
\frac{2}{3} \frac{V}{4 \pi^2 } 
\lr{ \frac{1}{b_{\mathrm{L}}^{3/2}} + \frac{2}{b_{\mathrm{T}}^{3/2}} }
\omega_{\mathrm{D}}^{3/2}
=
\frac{V}{6 \pi^2 } 
\lr{ \frac{1}{b_{\mathrm{L}}^{3/2}} + \frac{2}{b_{\mathrm{T}}^{3/2}} }
\omega_{\mathrm{D}}^{3/2}
\end{dmath}

\makeSubAnswer{}{pr:quadraticDebye:1:c}

Assuming a Bose distribution and ignoring the zero point energy, which has no temperature dependence, the specific heat, the temperature derivative of the energy density, is

\begin{dmath}\label{eqn:quadraticDebye:110}
\CV 
= \frac{d}{d T} \inv{V} \int Z(\omega) \frac{\Hbar \omega}{ e^{\Hbar \omega/ \kB T } - 1} d\omega
= \inv{V} \frac{d}{d T} \int Z(\omega) \frac{\Hbar \omega}{ \Hbar \omega/ \kB T + \inv{2}( \Hbar \omega/\kB T)^2 + \cdots } d\omega
\approx \inv{V} \frac{d}{d T} \int Z(\omega) \kB T d\omega
= \inv{V} \kB 3 r N.
\end{dmath}

\makeSubAnswer{}{pr:quadraticDebye:1:d}

First note that the density of states can be written

\begin{dmath}\label{eqn:quadraticDebye:130}
Z(\omega) = 
\frac{9 r N}{ 2 \omega_{\mathrm{D}}^{3/2} } \omega^{1/2},
\end{dmath}

for a specific heat of

\begin{dmath}\label{eqn:quadraticDebye:150}
\CV 
= \frac{d}{d T} \inv{V} \int_0^\infty \frac{9 r N}{ 2 \omega_{\mathrm{D}}^{3/2} } \omega^{1/2} \frac{\Hbar \omega}{ e^{\Hbar \omega/ \kB T } - 1} d\omega
= 
\frac{9 r N}{ 2 V \omega_{\mathrm{D}}^{3/2} } 
\int_0^\infty 
d\omega \omega^{1/2} 
\frac{d}{d T} 
\frac{\Hbar \omega}{ e^{\Hbar \omega/ \kB T } - 1} 
= 
\frac{9 r N}{ 2 V \omega_{\mathrm{D}}^{3/2} } 
\int_0^\infty 
d\omega \omega^{1/2} 
\frac{-\Hbar \omega}{ 
\lr{e^{\Hbar \omega/ \kB T } - 1}^2 
}  e^{\Hbar \omega/\kB T} \Hbar \omega/\kB 
\lr{-\inv{T^2}}
= 
\frac{9 r N \kB }{ 2 V \omega_{\mathrm{D}}^{3/2} } 
\lr{ \frac{ \kB T}{\Hbar} }
^{3/2}
\int_0^\infty 
d \frac{\Hbar \omega}{\kB T} 
\lr{\frac{\Hbar \omega}{\kB T}}
^{1/2} 
\frac{1}{ 
\lr{e^{\Hbar \omega/ \kB T } - 1}^2 
}  e^{\Hbar \omega/\kB T} 
\lr{ \frac{\Hbar \omega}{\kB T} }^2
= 
\frac{9 r N \kB }{ 2 V \omega_{\mathrm{D}}^{3/2} } 
\lr{ \frac{ \kB T}{\Hbar} }
^{3/2}
\int_0^\infty dy \frac{y^{5/2} e^y }{ 
\lr{e^y - 1}^2
 }
=
\frac{9 r N \kB }{ 2 V } 
\lr{ \frac{ T}{\Theta} }
^{3/2} \mathcal{I}.
\end{dmath}
} % makeanswer
